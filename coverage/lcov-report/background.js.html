const syncQuotes = (newArr, prevArr) => {
  if (!prevArr) {
    return newArr.map((qoute) => ({ ...qoute, show: true }));
  }
 
  return newArr.map((newQoute) => {
    const qoute = prevArr.find(
      (prevQoute) => prevQoute.qoute === newQoute.qoute
    );

    return { ...newQoute, show };
  });
};
